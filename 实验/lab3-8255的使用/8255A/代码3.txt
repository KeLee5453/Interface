#define  MY8255_B     IOY0 + 0x01*2
#define  MY8255_C     IOY0 + 0x02*2
#define  MY8255_MODE  IOY0 + 0x03*2

void main()
{
	char a;
	char buff[9];
	int b=0x00;
	outp(MY8255_MODE, 0x90);
	
	while(1)
	{
	  int i;
	   delay(5000);
		b=b+1;
		outp(MY8255_B, b);
		a = inp(MY8255_A);
		itoa(a,buff,2);
		printf("%08s\n",buff);
		
		//key();
	}
}

{
